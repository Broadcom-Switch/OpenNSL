18  * \file example_trunk.c
19  *
20  * \brief Example code for Trunk application
21  *
22  * \details This example demonstrates a simple trunk API scenario.
23  *
24  * This application also supports Warmboot feature. Warmboot is the
25  * process of restarting the device driver software while the hardware is
26  * running without interrupting the dataplane.
27  *
28  * Setup the following envinonment variable before running the application.
29  * For Cold boot mode, use "export OPENNSL_BOOT_FLAGS = 0x000000".
30  * For Warm boot mode, use "export OPENNSL_BOOT_FLAGS = 0x200000".
31  *
32  ************************************************************************/

88 int main(int argc, char *argv[])
89 {
90  opennsl_error_t rv;
91  opennsl_trunk_t tid;
92  opennsl_gport_t gport;
93  opennsl_port_t port;
94  int unit = DEFAULT_UNIT;
95  int member_max = MAX_TRUNK_MEMBER, member_count, i;
96  opennsl_trunk_member_t member;
97  opennsl_trunk_member_t member_arr[member_max];
98  opennsl_trunk_info_t t_data;
99  unsigned int warm_boot;
100  int choice;
101  int index = 0;
102 
103  if(strcmp(argv[0], "gdb") == 0)
104  {
105  index = 1;
106  }
107 
108  if((argc != (index + 1)) || ((argc > (index + 1)) && (strcmp(argv[index + 1], "--help") == 0))) {
109  printf("%s\n\r", example_usage);
110  return OPENNSL_E_PARAM;
111  }
112 
113  /* Initialize the system. */
114  printf("Initializing the system.\r\n");
115  rv = opennsl_driver_init((opennsl_init_t *) NULL);
116 
117  if(rv != OPENNSL_E_NONE) {
118  printf("\r\nFailed to initialize the system. Error %s\r\n",
119  opennsl_errmsg(rv));
120  return rv;
121  }
122 
123  warm_boot = opennsl_driver_boot_flags_get() & OPENNSL_BOOT_F_WARM_BOOT;
124 
125  if(!warm_boot)
126  {
127  /* Add ports to default vlan. */
128  printf("Adding ports to default vlan.\r\n");
129  rv = example_switch_default_vlan_config(unit);
130  if(rv != OPENNSL_E_NONE) {
131  printf("\r\nFailed to add default ports. rv: %s\r\n", opennsl_errmsg(rv));
132  return rv;
133  }
134  }
135 
136  while (1) {
137  printf("\r\nUser menu: Select one of the following options\r\n");
138  printf("1. Create trunk\n");
139  printf("2. Add port to trunk\n");
140  printf("3. Delete port from trunk\n");
141  printf("4. Get trunk info\n");
142  printf("5. Delete trunk\n");
143  printf("6. Save the configuration to scache\n");
144  printf("9. Launch diagnostic shell\n");
145  printf("0. Quit the application\n");
146 
147  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
148  {
149  printf("Invalid option entered. Please re-enter.\n");
150  continue;
151  }
152  switch(choice){
153 
154  case 1:
155  {
156  /* Create default trunk */
157  rv = example_trunk_create(unit, &tid);
158  if (rv == OPENNSL_E_NONE) {
159  printf("Trunk %d created successfully\n", tid);
160  } else {
161  printf("Failed to create trunk. Error %s\n",
162  opennsl_errmsg(rv));
163  continue;
164  }
165 
166  break;
167  } /* End of case 1 */
168 
169  case 2:
170  {
171  /* Add port to trunk*/
172  printf("\r\nEnter the port number.\r\n");
173  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
174  {
175  printf("Invalid option entered. Please re-enter.\n");
176  continue;
177  }
178  rv = opennsl_port_gport_get (unit, port, &gport);
179  if (rv != OPENNSL_E_NONE)
180  {
181  return OPENNSL_E_FAIL;
182  }
183 
184  opennsl_trunk_member_t_init(&member);
185  member.gport = gport;
186 
187  printf("\r\nEnter the trunk id.\r\n");
188  if(example_read_user_choice(&tid) != OPENNSL_E_NONE)
189  {
190  printf("Invalid option entered. Please re-enter.\n");
191  continue;
192  }
193 
194  printf("member_add: Adding port %d to trunk-id %d\n", port, tid);
195  rv = opennsl_trunk_member_add(unit, tid, &member);
196  if ((rv == OPENNSL_E_NONE)) {
197  printf("Port %d added successfully to trunk %d\n", port, tid);
198  } else {
199  printf("Failed to add port. Error %s\n",
200  opennsl_errmsg(rv));
201  continue;
202  }
203 
204  break;
205  } /* End of case 2 */
206 
207  case 3:
208  {
209  /* Delete port from trunk*/
210  printf("\r\nEnter the port number.\r\n");
211  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
212  {
213  printf("Invalid option entered. Please re-enter.\n");
214  continue;
215  }
216  rv = opennsl_port_gport_get (unit, port, &gport);
217  if (rv != OPENNSL_E_NONE)
218  {
219  return OPENNSL_E_FAIL;
220  }
221 
222  opennsl_trunk_member_t_init(&member);
223  member.gport = gport;
224 
225  printf("\r\nEnter the trunk id.\r\n");
226  if(example_read_user_choice(&tid) != OPENNSL_E_NONE)
227  {
228  printf("Invalid option entered. Please re-enter.\n");
229  continue;
230  }
231 
232  printf("member_delete: Deleting port %d from trunk-id %d\n", port, tid);
233  rv = opennsl_trunk_member_delete(unit, tid, &member);
234  if ((rv == OPENNSL_E_NONE)) {
235  printf("Port %d deleted successfully from trunk %d\n", port, tid);
236  } else {
237  printf("Failed to delete port. Error %s\n",
238  opennsl_errmsg(rv));
239  continue;
240  }
241 
242  break;
243  } /* End of case 3 */
244 
245  case 4:
246  {
247  /* Get trunk info*/
248  printf("\r\nEnter the trunk id.\r\n");
249  if(example_read_user_choice(&tid) != OPENNSL_E_NONE)
250  {
251  printf("Invalid option entered. Please re-enter.\n");
252  continue;
253  }
254 
255  for (i = 0; i < member_max; i++)
256  {
257  opennsl_trunk_member_t_init(&member_arr[i]);
258  }
259  opennsl_trunk_info_t_init(&t_data);
260 
261  rv = opennsl_trunk_get(unit, tid, &t_data, member_max,
262  member_arr, &member_count);
263  if ((rv != OPENNSL_E_NONE)) {
264  printf("Failed to get trunk info. Error %s\n",
265  opennsl_errmsg(rv));
266  continue;
267  }
268  printf("trunk %d has %d members\n",tid ,member_count);
269  for( i = 0; i < member_count; i++)
270  {
271  opennsl_port_local_get(unit, member_arr[i].gport, &port);
272  printf("port %d\n",port);
273  }
274 
275  break;
276  } /* End of case 4 */
277 
278  case 5:
279  {
280  /* Delete a trunk */
281  printf("\r\nEnter the trunk id.\r\n");
282  if(example_read_user_choice(&tid) != OPENNSL_E_NONE)
283  {
284  printf("Invalid option entered. Please re-enter.\n");
285  continue;
286  }
287 
288  rv = opennsl_trunk_destroy(unit, tid);
289  if ((rv == OPENNSL_E_NONE)) {
290  printf("trunk %d successfully deleted\n", tid);
291  } else {
292  printf("Failed to delete trunk. Error %s\n",
293  opennsl_errmsg(rv));
294  continue;
295  }
296 
297  break;
298  } /* End of case 5 */
299 
300  case 6:
301  {
302  /* Sync the current configuration */
303  rv = opennsl_switch_control_set(DEFAULT_UNIT, opennslSwitchControlSync, 1);
304  if(rv != OPENNSL_E_NONE) {
305  printf("Failed to synchronize the configuration to scache. "
306  "Error %s\n", opennsl_errmsg(rv));
307  return rv;
308  }
309  printf("Warmboot configuration is saved successfully.\n");
310  break;
311  } /* End of case 6 */
312 
313  case 9:
314  {
315  opennsl_driver_shell();
316  break;
317  }
318 
319  case 0:
320  {
321  printf("Exiting the application.\n");
322  return OPENNSL_E_NONE;
323  }
324 
325  default:
326  break;
327  } /* End of switch */
328  } /* End of while */
329 
330  return OPENNSL_E_NONE;
331 }
