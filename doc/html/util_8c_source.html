77 int example_read_user_choice(int *choice)
78 {
79  char val;
80  char digits[MAX_DIGITS_IN_CHOICE + 1];
81  int idx = 0;
82  int valid = TRUE;
83 
84  /* parse input string until \n */
85  while((val = getchar()) != '\n')
86  {
87  if ((val >= '0' && val <= '9') && idx < MAX_DIGITS_IN_CHOICE)
88  {
89  digits[idx++] = val;
90  }
91  else
92  {
93  valid = FALSE;
94  }
95  }
96  if ((valid == TRUE) && idx != 0)
97  {
98  digits[idx] = '\0';
99  *choice = atoi(digits);
100  return(OPENNSL_E_NONE);
101  }
102  else
103  {
104  *choice = -1;
105  return(OPENNSL_E_FAIL);
106  }

118 int opennsl_mac_parse(char *buf, unsigned char *macp)
119 {
120  int i, c1, c2;
121  char *s;
122 #define MAC_ADDR_LEN 17
123 
124  macp[0] = macp[1] = macp[2] = 0;
125  macp[3] = macp[4] = macp[5] = 0;
126 
127  if ((buf == NULL) || (strlen(buf) > MAC_ADDR_LEN)) {
128  return OPENNSL_E_FAIL;
129  }
130 
131  /* skip leading 0x if plain hex format */
132  if (buf[0] == '0' && (buf[1] == 'x' || buf[1] == 'X')) {
133  buf += 2;
134  }
135 
136  /* start at end of string and work backwards */
137  for (s = buf; *s; s++) {
138  ;
139  }
140  for (i = 5; i >= 0 && s >= buf; i--) {
141  c1 = c2 = 0;
142  if (--s >= buf) {
143  if (*s >= '0' && *s <= '9') {
144  c2 = *s - '0';
145  } else if (*s >= 'a' && *s <= 'f') {
146  c2 = *s - 'a' + 10;
147  } else if (*s >= 'A' && *s <= 'F') {
148  c2 = *s - 'A' + 10;
149  } else if (*s == ':') {
150  ;
151  } else {
152  return OPENNSL_E_FAIL;
153  }
154  }
155  if (*s != ':' && --s >= buf) {
156  if (*s >= '0' && *s <= '9') {
157  c1 = *s - '0';
158  } else if (*s >= 'a' && *s <= 'f') {
159  c1 = *s - 'a' + 10;
160  } else if (*s >= 'A' && *s <= 'F') {
161  c1 = *s - 'A' + 10;
162  } else if (*s == ':') {
163  ;
164  } else {
165  return OPENNSL_E_FAIL;
166  }
167  }
168 
169  if (s > buf && s[-1] == ':') {
170  --s;
171  }
172  macp[i] = c1 << 4 | c2;
173  }
174  return OPENNSL_E_NONE;
175 }

207 int opennsl_ip_parse(char *ip_str, unsigned int *ip_val)
208 {
209  unsigned int num = 0, val;
210  char *tok;
211  int count = 0;
212  char buf[16]; /* Maximum length of IP address in dotten notation */
213 
214  if((ip_str == NULL) || (ip_val == NULL) || (strlen(ip_str) > 16))
215  {
216  return -1;
217  }
218  strcpy(buf, ip_str);
219  tok = strtok(buf, ".");
220  while(tok != NULL)
221  {
222  count++;
223  val = atoi(tok);
224  if((val < 0) || (val > 0xff))
225  {
226  return -1;
227  }
228  num = (num << 8) + val;
229  tok = strtok(NULL, ".");
230  }
231  if(count != 4)
232  {
233  return -1;
234  }
235  *ip_val = num;
236 
237  return 0;
238 }

271 int opennsl_ctoi(const char *s, char **end)
272 {
273  unsigned int n, neg;
274  int base = 10;
275 
276  if (s == 0) {
277  if (end != 0) {
278  end = 0;
279  }
280  return 0;
281  }
282 
283  s += (neg = (*s == '-'));
284 
285  if (*s == '0') {
286  s++;
287  if (*s == 'x' || *s == 'X') {
288  base = 16;
289  s++;
290  } else if (*s == 'b' || *s == 'B') {
291  base = 2;
292  s++;
293  } else {
294  base = 8;
295  }
296  }
297 
298  for (n = 0; ((*s >= 'a' && *s < 'a' + base - 10) ||
299  (*s >= 'A' && *s < 'A' + base - 10) ||
300  (*s >= '0' && *s <= '9')); s++) {
301  n = n * base + ((*s <= '9' ? *s : *s + 9) & 15);
302  }
303 
304  if (end != 0) {
305  *end = (char *) s;
306  }
307 
308  return (int) (neg ? -n : n);
309 }
