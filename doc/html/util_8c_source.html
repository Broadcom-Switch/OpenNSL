106 {

110  int valid = TRUE;
111 
112  /* parse input string until \n */
113  while((val = getchar()) != '\n')
114  {
115  if ((val >= '0' && val <= '9') && idx < MAX_DIGITS_IN_CHOICE)
116  {
117  digits[idx++] = val;
118  }
119  else
120  {
121  valid = FALSE;
122  }
123  }
124  if ((valid == TRUE) && idx != 0)
125  {
126  digits[idx] = '\0';
127  *choice = atoi(digits);
128  return(OPENNSL_E_NONE);
129  }
130  else
131  {
132  *choice = -1;
133  return(OPENNSL_E_FAIL);

146 int opennsl_mac_parse(char *buf, unsigned char *macp)
147 {
148  int i, c1, c2;
149  char *s;
150 #define MAC_ADDR_LEN 17
151 
152  macp[0] = macp[1] = macp[2] = 0;
153  macp[3] = macp[4] = macp[5] = 0;
154 
155  if ((buf == NULL) || (strlen(buf) > MAC_ADDR_LEN)) {
156  return OPENNSL_E_FAIL;
157  }
158 
159  /* skip leading 0x if plain hex format */
160  if (buf[0] == '0' && (buf[1] == 'x' || buf[1] == 'X')) {
161  buf += 2;
162  }
163 
164  /* start at end of string and work backwards */
165  for (s = buf; *s; s++) {
166  ;
167  }
168  for (i = 5; i >= 0 && s >= buf; i--) {
169  c1 = c2 = 0;
170  if (--s >= buf) {
171  if (*s >= '0' && *s <= '9') {
172  c2 = *s - '0';
173  } else if (*s >= 'a' && *s <= 'f') {
174  c2 = *s - 'a' + 10;
175  } else if (*s >= 'A' && *s <= 'F') {
176  c2 = *s - 'A' + 10;
177  } else if (*s == ':') {
178  ;
179  } else {
180  return OPENNSL_E_FAIL;
181  }
182  }
183  if (*s != ':' && --s >= buf) {
184  if (*s >= '0' && *s <= '9') {
185  c1 = *s - '0';
186  } else if (*s >= 'a' && *s <= 'f') {
187  c1 = *s - 'a' + 10;
188  } else if (*s >= 'A' && *s <= 'F') {
189  c1 = *s - 'A' + 10;
190  } else if (*s == ':') {
191  ;
192  } else {
193  return OPENNSL_E_FAIL;
194  }
195  }
196 
197  if (s > buf && s[-1] == ':') {
198  --s;
199  }
200  macp[i] = c1 << 4 | c2;
201  }
202  return OPENNSL_E_NONE;
203 }

235 int opennsl_ip_parse(char *ip_str, unsigned int *ip_val)
236 {
237  unsigned int num = 0, val;
238  char *tok;
239  int count = 0;
240  char buf[16]; /* Maximum length of IP address in dotten notation */
241 
242  if((ip_str == NULL) || (ip_val == NULL) || (strlen(ip_str) > 16))
243  {
244  return -1;
245  }
246  strcpy(buf, ip_str);
247  tok = strtok(buf, ".");
248  while(tok != NULL)
249  {
250  count++;
251  val = atoi(tok);
252  if((val < 0) || (val > 0xff))
253  {
254  return -1;
255  }
256  num = (num << 8) + val;
257  tok = strtok(NULL, ".");
258  }
259  if(count != 4)
260  {
261  return -1;
262  }
263  *ip_val = num;
264 
265  return 0;
266 }

299 int opennsl_ctoi(const char *s, char **end)
300 {
301  unsigned int n, neg;
302  int base = 10;
303 
304  if (s == 0) {
305  if (end != 0) {
306  end = 0;
307  }
308  return 0;
309  }
310 
311  s += (neg = (*s == '-'));
312 
313  if (*s == '0') {
314  s++;
315  if (*s == 'x' || *s == 'X') {
316  base = 16;
317  s++;
318  } else if (*s == 'b' || *s == 'B') {
319  base = 2;
320  s++;
321  } else {
322  base = 8;
323  }
324  }
325 
326  for (n = 0; ((*s >= 'a' && *s < 'a' + base - 10) ||
327  (*s >= 'A' && *s < 'A' + base - 10) ||
328  (*s >= '0' && *s <= '9')); s++) {
329  n = n * base + ((*s <= '9' ? *s : *s + 9) & 15);
330  }
331 
332  if (end != 0) {
333  *end = (char *) s;
334  }
335 
336  return (int) (neg ? -n : n);
337 }
