48 int example_port_default_config(int unit)
49 {
50  opennsl_port_config_t pcfg;
51  opennsl_port_info_t info;
52  int rv;
53  int port;
54  int stp_state = OPENNSL_STG_STP_FORWARD;
55  int stg = 1;
56 
57  /*
58  * Create VLAN with id DEFAULT_VLAN and
59  * add ethernet ports to the VLAN
60  */
61  rv = opennsl_port_config_get(unit, &pcfg);
62  if (rv != OPENNSL_E_NONE)
63  {
64  printf("Failed to get port configuration. Error %s\n", opennsl_errmsg(rv));
65  return rv;
66  }
67 
68  /* Set the STP state to forward in default STG for all ports */
69  OPENNSL_PBMP_ITER(pcfg.e, port)
70  {
71  rv = opennsl_stg_stp_set(unit, stg, port, stp_state);
72  if (rv != OPENNSL_E_NONE)
73  {
74  printf("Failed to set STP state for unit %d port %d, Error %s\n",
75  unit, port, opennsl_errmsg(rv));
76  return rv;
77  }
78  }
79 
80  /* Setup default configuration on the ports */
81  opennsl_port_info_t_init(&info);
82 
83  info.speed = 0;
84  info.duplex = OPENNSL_PORT_DUPLEX_FULL;
85  info.pause_rx = OPENNSL_PORT_ABILITY_PAUSE_RX;
86  info.pause_tx = OPENNSL_PORT_ABILITY_PAUSE_TX;
87  info.linkscan = OPENNSL_LINKSCAN_MODE_SW;
88  info.autoneg = FALSE;
89  info.enable = 1;
90 
91  info.action_mask |= ( OPENNSL_PORT_ATTR_AUTONEG_MASK |
92  OPENNSL_PORT_ATTR_SPEED_MASK |
93  OPENNSL_PORT_ATTR_DUPLEX_MASK |
94  OPENNSL_PORT_ATTR_PAUSE_RX_MASK |
95  OPENNSL_PORT_ATTR_PAUSE_TX_MASK |
96  OPENNSL_PORT_ATTR_LINKSCAN_MASK |
97  OPENNSL_PORT_ATTR_ENABLE_MASK );
98 
99  OPENNSL_PBMP_ITER(pcfg.e, port)
100  {
101  rv = opennsl_port_selective_set(unit, port, &info);
102  if (OPENNSL_FAILURE(rv))
103  {
104  printf("Failed to set port config for unit %d, port %d, Error %s",
105  unit, port, opennsl_errmsg(rv));
106  return rv;
107  }
108  }
109 
110  return OPENNSL_E_NONE;
111 }

181 int example_read_user_choice(int *choice)
182 {
183  char val;
184  char digits[MAX_DIGITS_IN_CHOICE + 1];
185  int idx = 0;
186  int valid = TRUE;
187 
188  /* parse input string until \n */
189  while((val = getchar()) != '\n')
190  {
191  if ((val >= '0' && val <= '9') && idx < MAX_DIGITS_IN_CHOICE)
192  {
193  digits[idx++] = val;
194  }
195  else
196  {
197  valid = FALSE;
198  }
199  }
200  if ((valid == TRUE) && idx != 0)
201  {
202  digits[idx] = '\0';
203  *choice = atoi(digits);
204  return(OPENNSL_E_NONE);
205  }
206  else
207  {
208  *choice = -1;
209  return(OPENNSL_E_FAIL);
210  }
211 }

222 int opennsl_mac_parse(char *buf, unsigned char *macp)
223 {
224  int i, c1, c2;
225  char *s;
226 #define MAC_ADDR_LEN 17
227 
228  macp[0] = macp[1] = macp[2] = 0;
229  macp[3] = macp[4] = macp[5] = 0;
230 
231  if ((buf == NULL) || (strlen(buf) > MAC_ADDR_LEN)) {
232  return OPENNSL_E_FAIL;
233  }
234 
235  /* skip leading 0x if plain hex format */
236  if (buf[0] == '0' && (buf[1] == 'x' || buf[1] == 'X')) {
237  buf += 2;
238  }
239 
240  /* start at end of string and work backwards */
241  for (s = buf; *s; s++) {
242  ;
243  }
244  for (i = 5; i >= 0 && s >= buf; i--) {
245  c1 = c2 = 0;
246  if (--s >= buf) {
247  if (*s >= '0' && *s <= '9') {
248  c2 = *s - '0';
249  } else if (*s >= 'a' && *s <= 'f') {
250  c2 = *s - 'a' + 10;
251  } else if (*s >= 'A' && *s <= 'F') {
252  c2 = *s - 'A' + 10;
253  } else if (*s == ':') {
254  ;
255  } else {
256  return OPENNSL_E_FAIL;
257  }
258  }
259  if (*s != ':' && --s >= buf) {
260  if (*s >= '0' && *s <= '9') {
261  c1 = *s - '0';
262  } else if (*s >= 'a' && *s <= 'f') {
263  c1 = *s - 'a' + 10;
264  } else if (*s >= 'A' && *s <= 'F') {
265  c1 = *s - 'A' + 10;
266  } else if (*s == ':') {
267  ;
268  } else {
269  return OPENNSL_E_FAIL;
270  }
271  }
272 
273  if (s > buf && s[-1] == ':') {
274  --s;
275  }
276  macp[i] = c1 << 4 | c2;
277  }
278  return OPENNSL_E_NONE;
279 }

311 int opennsl_ip_parse(char *ip_str, unsigned int *ip_val)
312 {
313  unsigned int num = 0, val;
314  char *tok;
315  int count = 0;
316  char buf[16]; /* Maximum length of IP address in dotten notation */
317 
318  if((ip_str == NULL) || (ip_val == NULL) || (strlen(ip_str) > 16))
319  {
320  return -1;
321  }
322  strcpy(buf, ip_str);
323  tok = strtok(buf, ".");
324  while(tok != NULL)
325  {
326  count++;
327  val = atoi(tok);
328  if((val < 0) || (val > 0xff))
329  {
330  return -1;
331  }
332  num = (num << 8) + val;
333  tok = strtok(NULL, ".");
334  }
335  if(count != 4)
336  {
337  return -1;
338  }
339  *ip_val = num;
340 
341  return 0;
342 }

375 int opennsl_ctoi(const char *s, char **end)
376 {
377  unsigned int n, neg;
378  int base = 10;
379 
380  if (s == 0) {
381  if (end != 0) {
382  end = 0;
383  }
384  return 0;
385  }
386 
387  s += (neg = (*s == '-'));
388 
389  if (*s == '0') {
390  s++;
391  if (*s == 'x' || *s == 'X') {
392  base = 16;
393  s++;
394  } else if (*s == 'b' || *s == 'B') {
395  base = 2;
396  s++;
397  } else {
398  base = 8;
399  }
400  }
401 
402  for (n = 0; ((*s >= 'a' && *s < 'a' + base - 10) ||
403  (*s >= 'A' && *s < 'A' + base - 10) ||
404  (*s >= '0' && *s <= '9')); s++) {
405  n = n * base + ((*s <= '9' ? *s : *s + 9) & 15);
406  }
407 
408  if (end != 0) {
409  *end = (char *) s;
410  }
411 
412  return (int) (neg ? -n : n);
413 }
