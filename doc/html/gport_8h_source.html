19  * Details: This file defines common error codes to be shared

27 #ifndef _SHR_GPORT_H
28 #define _SHR_GPORT_H
29 
30 /*
31  * Defines:
32  * _SHR_GPORT_*
33  * Purpose:
34  * GPORT (Generic port) definitions. GPORT can be used to specify
35  * a {module, port} pair, trunk, and other port types.
36  */
37 
38 #define _SHR_GPORT_NONE (0)
39 #define _SHR_GPORT_INVALID (-1)
40 #define _SHR_GPORT_TYPE_LOCAL 1 /* Port on local unit */
41 #define _SHR_GPORT_TYPE_MODPORT 2 /* Module ID and port */
42 #define _SHR_GPORT_TYPE_TRUNK 3 /* Trunk ID */
43 #define _SHR_GPORT_TYPE_LOCAL_CPU 5 /* CPU destination */
44 #define _SHR_GPORT_TYPE_UCAST_QUEUE_GROUP 9 /* Queue Group */
45 #define _SHR_GPORT_TYPE_EGRESS_GROUP 21 /* Egress group port */
46 #define _SHR_GPORT_TYPE_EGRESS_CHILD 22 /* Egress child port */
47 #define _SHR_GPORT_TYPE_EGRESS_MODPORT 23 /* Egress Mod and port */
48 
49 #define _SHR_GPORT_TYPE_SHIFT 26
50 #define _SHR_GPORT_TYPE_MASK 0x3f
51 
52 #define _SHR_GPORT_MODID_SHIFT 11
53 #define _SHR_GPORT_MODID_MASK 0x7fff
54 #define _SHR_GPORT_PORT_SHIFT 0
55 #define _SHR_GPORT_PORT_MASK 0x7ff
56 #define _SHR_GPORT_TRUNK_SHIFT 0
57 #define _SHR_GPORT_TRUNK_MASK 0x3ffffff
58 
59 #define _SHR_GPORT_IS_MODPORT(_gport) \
60  (((_gport) >> _SHR_GPORT_TYPE_SHIFT) == _SHR_GPORT_TYPE_MODPORT)
61 
62 #define _SHR_GPORT_MODPORT_SET(_gport, _module, _port) \
63  ((_gport) = (_SHR_GPORT_TYPE_MODPORT << _SHR_GPORT_TYPE_SHIFT) | \
64  (((_module) & _SHR_GPORT_MODID_MASK) << _SHR_GPORT_MODID_SHIFT) | \
65  (((_port) & _SHR_GPORT_PORT_MASK) << _SHR_GPORT_PORT_SHIFT))
66 
67 #define _SHR_GPORT_MODPORT_MODID_GET(_gport) \
68  (((_gport) >> _SHR_GPORT_MODID_SHIFT) & _SHR_GPORT_MODID_MASK)
69 
70 #define _SHR_GPORT_MODPORT_PORT_GET(_gport) \
71  (((_gport) >> _SHR_GPORT_PORT_SHIFT) & _SHR_GPORT_PORT_MASK)
72 
73 #define _SHR_GPORT_IS_TRUNK(_gport) \
74  (((_gport) >> _SHR_GPORT_TYPE_SHIFT) == _SHR_GPORT_TYPE_TRUNK)
75 
76 #define _SHR_GPORT_TRUNK_SET(_gport, _trunk_id) \
77  ((_gport) = (_SHR_GPORT_TYPE_TRUNK << _SHR_GPORT_TYPE_SHIFT) | \
78  (((_trunk_id) & _SHR_GPORT_TRUNK_MASK) << _SHR_GPORT_TRUNK_SHIFT))
79 
80 #define _SHR_GPORT_TRUNK_GET(_gport) \
81  (((_gport) >> _SHR_GPORT_TRUNK_SHIFT) & _SHR_GPORT_TRUNK_MASK)
82 
83 #define _SHR_GPORT_LOCAL_CPU \
84  (_SHR_GPORT_TYPE_LOCAL_CPU << _SHR_GPORT_TYPE_SHIFT)
85 
86 #endif /* !_SHR_GPORT_H */
