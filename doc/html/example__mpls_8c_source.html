00206 static int mpls_lsr_term(int unit, int in_port, int out_port)
00207 {
00208 
00209   opennsl_vpn_t vpn_type_id;
00210   opennsl_mpls_vpn_config_t vpn_info;
00211   opennsl_pbmp_t pbmp, ubmp;
00212   opennsl_mpls_tunnel_switch_t tunnel_switch;
00213   opennsl_l3_intf_t l3_intf;
00214   opennsl_mac_t mac_sa = MAC_SA;
00215   opennsl_l3_egress_t l3_egress;
00216   opennsl_gport_t  gport = OPENNSL_GPORT_INVALID;
00217   opennsl_mac_t mac_da = MAC_DA;
00218   opennsl_mac_t local_mac = MY_MAC;
00219   opennsl_if_t egr_obj_1;
00220   opennsl_mpls_egress_label_t tun_label;
00221 
00222   CALL_IF_ERROR_RETURN(opennsl_switch_control_set(unit,
00223         opennslSwitchL3EgressMode, 1));
00224 
00225   CALL_IF_ERROR_RETURN(opennsl_port_control_set(unit, out_port,
00226         opennslPortControlDoNotCheckVlan, 1));
00227 
00228   CALL_IF_ERROR_RETURN(opennsl_port_control_set(unit, in_port,
00229         opennslPortControlDoNotCheckVlan, 1));
00230 
00231   /*  Create L3 MPLS vpn */
00232   opennsl_mpls_vpn_config_t_init(&vpn_info);
00233   vpn_info.flags = OPENNSL_MPLS_VPN_L3 | OPENNSL_MPLS_VPN_WITH_ID;
00234   vpn_info.vpn = VPN_ID;
00235   vpn_info.lookup_id = 0;
00236   CALL_IF_ERROR_RETURN(opennsl_mpls_vpn_id_create(unit, &vpn_info));
00237   vpn_type_id = vpn_info.vpn;
00238 
00239   if(verbose >= 2) {
00240     printf("VPN is created with ID: %d\n", vpn_type_id);
00241   }
00242 
00243   /*  Create Incoming VLAN and ingress port to the incoming VLAN */
00244   opennsl_vlan_create(unit, IN_VLAN);
00245   OPENNSL_PBMP_CLEAR(ubmp);
00246   OPENNSL_PBMP_CLEAR(pbmp);
00247   OPENNSL_PBMP_PORT_ADD(pbmp, in_port);
00248   CALL_IF_ERROR_RETURN(opennsl_vlan_port_add(unit, IN_VLAN, pbmp, ubmp));
00249 
00250   if(verbose >= 2) {
00251     printf("Port %d is added to Ingress VLAN %d\n", in_port, IN_VLAN);
00252   }
00253 
00254   /*  Create egress VLAN and egress port to the egress VLAN */
00255   opennsl_vlan_create(unit, OUT_VLAN);
00256   OPENNSL_PBMP_CLEAR(ubmp);
00257   OPENNSL_PBMP_CLEAR(pbmp);
00258   OPENNSL_PBMP_PORT_ADD(pbmp, out_port);
00259   CALL_IF_ERROR_RETURN(opennsl_vlan_port_add(unit, OUT_VLAN, pbmp, ubmp));
00260 
00261   if(verbose >= 2) {
00262     printf("Port %d is added to Egress VLAN %d\n", out_port, OUT_VLAN);
00263   }
00264 
00265   /*  Create outgoing L3 interface */
00266   opennsl_l3_intf_t_init(&l3_intf);
00267   l3_intf.l3a_flags  = OPENNSL_L3_WITH_ID;
00268   l3_intf.l3a_intf_id = L3_OUT_INTF_ID;
00269   l3_intf.l3a_vid = OUT_VLAN;
00270   l3_intf.l3a_vrf = VRF_ID;
00271   memcpy(l3_intf.l3a_mac_addr, mac_sa, sizeof(opennsl_mac_t));
00272   CALL_IF_ERROR_RETURN(opennsl_l3_intf_create(unit, &l3_intf));
00273 
00274   if(verbose >= 2) {
00275     printf("\r\nOutgoing L3 interface %d is created with the following "
00276         "parameters\n", l3_intf.l3a_intf_id);
00277     printf("    Egress VLAN %d\n", OUT_VLAN);
00278     l2_print_mac("    MAC address:", mac_sa);
00279     printf("\r\n");
00280   }
00281 
00282   /* Create outgoing egress object */
00283   opennsl_l3_egress_t_init(&l3_egress);
00284   CALL_IF_ERROR_RETURN(opennsl_port_gport_get(unit, out_port, &gport));
00285   l3_egress.intf     =  L3_OUT_INTF_ID;
00286   memcpy(l3_egress.mac_addr, mac_da , sizeof(opennsl_mac_t));
00287   l3_egress.vlan     =  OUT_VLAN;
00288   l3_egress.port     =  gport;
00289   CALL_IF_ERROR_RETURN(opennsl_l3_egress_create(unit, 0, &l3_egress,
00290         &egr_obj_1));
00291   if(verbose >= 2) {
00292     printf("\r\n");
00293     printf("Outgoing Egress object %d is created with the "
00294         "following parameters\n", egr_obj_1);
00295     printf("    Port %d Egress VLAN %d L3 interface %d\n", OUT_VLAN,
00296         gport, L3_OUT_INTF_ID);
00297     l2_print_mac("    Next hop MAC address:", mac_da);
00298     printf("\r\n");
00299   }
00300 
00301   /*  Add Entry for Incoming label Matching and SWAP */
00302   opennsl_mpls_tunnel_switch_t_init(&tunnel_switch);
00303   tunnel_switch.label  =  INGRESS_LABEL1_ID;
00304   tunnel_switch.port   =  OPENNSL_GPORT_INVALID;
00305   tunnel_switch.action =  OPENNSL_MPLS_SWITCH_ACTION_SWAP;
00306   tunnel_switch.vpn =     vpn_type_id;
00307   tunnel_switch.egress_if = egr_obj_1;
00308   tunnel_switch.egress_label.flags = OPENNSL_MPLS_EGRESS_LABEL_TTL_SET;
00309   tunnel_switch.egress_label.label = EGRESS_LABEL1_ID;
00310   tunnel_switch.egress_label.exp = 0;
00311   tunnel_switch.egress_label.ttl = LABEL1_TTL;
00312   tunnel_switch.egress_label.pkt_pri = 0;
00313   tunnel_switch.egress_label.pkt_cfi = 0;
00314   CALL_IF_ERROR_RETURN(opennsl_mpls_tunnel_switch_add(unit, &tunnel_switch));
00315 
00316   if(verbose >= 2) {
00317     printf("\r\n");
00318     printf("Added entry for incoming label matching and Swap with the "
00319         "following parameters\n");
00320     printf("    Incoming label %d (0x%x)\n", INGRESS_LABEL1_ID,
00321         INGRESS_LABEL1_ID);
00322     printf("    Action SWAP\n");
00323     printf("    Egress label: %d (0x%x) TTL: %d\n", EGRESS_LABEL1_ID,
00324         EGRESS_LABEL1_ID,
00325         LABEL1_TTL);
00326   }
00327 
00328   /* Install L2 tunnel MAC */
00329   CALL_IF_ERROR_RETURN(opennsl_l2_tunnel_add(unit, local_mac, IN_VLAN));
00330 
00331   if(verbose >= 2) {
00332     printf("\n\r");
00333     printf("Installed L2 tunnel for Ingress VLAN %d ", IN_VLAN);
00334     l2_print_mac("MAC address:", local_mac);
00335     printf("\n\r");
00336   }
00337 
00338   /* Add MPLS label to packet */
00339   opennsl_mpls_egress_label_t_init(&tun_label);
00340   tun_label.flags   =   OPENNSL_MPLS_EGRESS_LABEL_TTL_SET;
00341   tun_label.label   =   EGRESS_LABEL2_ID;
00342   tun_label.ttl     =   LABEL2_TTL;
00343   CALL_IF_ERROR_RETURN(opennsl_mpls_tunnel_initiator_set(unit, L3_OUT_INTF_ID,
00344         1, &tun_label));
00345   if(verbose >= 2) {
00346     printf("Updated tunnel to add MPLS label %d with TTL %d to the packet\n",
00347         tun_label.label, tun_label.ttl);
00348   }
00349 
00350   return OPENNSL_E_NONE;
00351 }
