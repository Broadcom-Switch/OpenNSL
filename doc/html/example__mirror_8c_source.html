00173 static int example_mirror_setup(int unit, int mode, int srcPort, int probePort)
00174 {
00175   int rv;
00176   opennsl_gport_t gport;
00177   unsigned int flags = 0;
00178   opennsl_mirror_destination_t mirror_dest;
00179   
00180   rv = opennsl_mirror_init(unit);
00181   if(rv != OPENNSL_E_NONE) {
00182     printf("\r\nFailed to init mirror module. Error: %s\r\n",
00183         opennsl_errmsg(rv));
00184   }
00185 
00186   opennsl_mirror_destination_t_init(&mirror_dest);
00187   OPENNSL_GPORT_MODPORT_SET(gport, 0, probePort);
00188   mirror_dest.gport = gport;
00189 
00190   rv = opennsl_mirror_destination_create(unit, &mirror_dest);
00191   if(rv != OPENNSL_E_NONE) {
00192     printf("\r\nFailed to create mirror dest. Error: %s\r\n",
00193         opennsl_errmsg(rv));
00194   }
00195 
00196   /* Configure the source port and mirror to port along with mirror mode */
00197   switch (mode)
00198   {
00199     case 1:
00200       /* Ingress mirroring only */
00201       flags |= OPENNSL_MIRROR_PORT_INGRESS;
00202       break;
00203 
00204     case 2:
00205       /* Egress mirroring only */
00206       flags |= OPENNSL_MIRROR_PORT_EGRESS;
00207       break;
00208 
00209     case 3:
00210       /* Both Ingress and Egress mirroring */
00211       flags |= (OPENNSL_MIRROR_PORT_INGRESS | OPENNSL_MIRROR_PORT_EGRESS);
00212       break;
00213 
00214     default:
00215       break;
00216   }
00217 
00218   rv = opennsl_mirror_port_dest_add(unit, srcPort, flags, mirror_dest.mirror_dest_id);
00219   if(rv != OPENNSL_E_NONE) {
00220     printf("\r\nFailed to enable mirroring on the port. Error: %s\r\n",
00221         opennsl_errmsg(rv));
00222   }
00223 
00224   return rv;
00225 }
