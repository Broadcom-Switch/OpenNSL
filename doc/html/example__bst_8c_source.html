87 int example_bst_default_profile_set (int unit)
88 {
89  int rc;
90  int bid;
91  int depth;
92  int index;
93  int max_threshold;
94  int max_ports;
95  int gport;
96  int port;
97  opennsl_cosq_bst_profile_t profile;
98 
99  /* Get maximum number of front panel ports available */
100  if((rc = example_max_port_count_get(unit, &max_ports)) != OPENNSL_E_NONE)
101  {
102  printf("\r\nFailed to get number of front panel ports, rc = %d (%s).\r\n",
103  rc, opennsl_errmsg(rc));
104  return OPENNSL_E_FAIL;
105  }
106 
107  for (bid = 0; bid < opennslBstStatIdMaxCount; bid++) {
108 
109  /* Find out the maximum threshold for a bid */
110  if (bid == opennslBstStatIdUcast ||
111  bid == opennslBstStatIdEgrUCastPortShared ||
112  bid == opennslBstStatIdUcastGroup) {
113  max_threshold = 0x3FFF * 208;
114  } else if (bid == opennslBstStatIdPriGroupHeadroom) {
115  max_threshold = 0xFFF * 208;
116  } else {
117  max_threshold = 0x1FFFF * 208;
118  }
119 
120  profile.byte = max_threshold;
121 
122  if (bid == _BST_STAT_ID_EGR_POOL ||
123  bid == _BST_STAT_ID_EGR_MCAST_POOL ||
124  bid == _BST_STAT_ID_EGR_UCAST_PORT_SHARED ||
125  bid == _BST_STAT_ID_EGR_PORT_SHARED ||
126  bid == _BST_STAT_ID_UCAST_GROUP ||
127  bid == _BST_STAT_ID_PORT_POOL ||
128  bid == _BST_STAT_ID_ING_POOL)
129  {
130  depth = 4;
131  }
132  else if (bid == _BST_STAT_ID_PRI_GROUP_SHARED ||
133  bid == _BST_STAT_ID_PRI_GROUP_HEADROOM ||
134  bid == _BST_STAT_ID_UCAST ||
135  bid == _BST_STAT_ID_MCAST)
136  {
137  depth = 8;
138  }
139  else if (bid == _BST_STAT_ID_RQE_QUEUE)
140  {
141  depth = 11;
142  }
143  else
144  {
145  depth = 1;
146  }
147 
148  /* Set default BST profiles */
149  if (bid == _BST_STAT_ID_ING_POOL ||
150  bid == _BST_STAT_ID_EGR_MCAST_POOL ||
151  bid == _BST_STAT_ID_EGR_POOL ||
152  bid == _BST_STAT_ID_RQE_QUEUE ||
153  bid == _BST_STAT_ID_UCAST_GROUP ||
154  bid == _BST_STAT_ID_DEVICE)
155  {
156  gport = 0;
157  for (index = 0; index < depth; index++)
158  {
159  rc = opennsl_cosq_bst_profile_set (unit, gport, index, bid, &profile);
160  if (rc != OPENNSL_E_NONE)
161  {
162  printf("\r\nFailed to set the BST profile for gport %d bid %d index %d, rc = %d (%s).\r\n",
163  gport, bid, index, rc, opennsl_errmsg(rc));
164  return rc;
165  }
166  } /* Iter ... Cosq*/
167  }
168  else if (bid == _BST_STAT_ID_PRI_GROUP_SHARED ||
169  bid == _BST_STAT_ID_PRI_GROUP_HEADROOM ||
170  bid == _BST_STAT_ID_UCAST ||
171  bid == _BST_STAT_ID_MCAST ||
172  bid == _BST_STAT_ID_PORT_POOL ||
173  bid == _BST_STAT_ID_EGR_PORT_SHARED ||
174  bid == _BST_STAT_ID_EGR_UCAST_PORT_SHARED)
175  {
176  for(port = 1; port < max_ports; port++)
177  {
178  /* Get GPORT*/
179  rc = opennsl_port_gport_get (unit, port, &gport);
180  if (rc != OPENNSL_E_NONE)
181  {
182  printf("\r\nFailed to get gport for port %d, rc = %d (%s).\r\n",
183  port, rc, opennsl_errmsg(rc));
184  return rc;
185  }
186 
187  for (index = 0; index < depth; index++)
188  {
189  rc = opennsl_cosq_bst_profile_set (unit, gport, index, bid, &profile);
190  if (rc != OPENNSL_E_NONE)
191  {
192  printf("\r\nFailed to set the BST profile for gport %d bid %d index %d, rc = %d (%s).\r\n",
193  gport, bid, index, rc, opennsl_errmsg(rc));
194  return rc;
195  }
196  } /* Iter ... index */
197  } /* Iter ... port */
198  }
199 
200  } /* Iter ... bid */
201 
202  return 0;
203 }

234 /* Main function for BST application */
235 int main(int argc, char *argv[])
236 {
237  opennsl_error_t rc;
238  int choice;
239  int unit = DEFAULT_UNIT;
240  int index,i,j;
241  opennsl_gport_t gport;
242  opennsl_port_t port;
243  opennsl_cos_queue_t cosq;
244  uint32 options = 0;
245  void *user_data = NULL;
246 
247  example_bst_counter_t id_list[MAX_COUNTERS] = {
248  {opennslBstStatIdUcast, "opennslBstStatIdUcast"},
249  {opennslBstStatIdMcast, "opennslBstStatIdMcast"},
250  {opennslBstStatIdPriGroupShared, "opennslBstStatIdPriGroupShared"},
251  {opennslBstStatIdPriGroupHeadroom, "opennslBstStatIdPriGroupHeadroom"}
252  };
253  uint64 val[MAX_COUNTERS][MAX_COSQ_COUNT];
254 
255  /* Intitializing val array */
256  for (i = 0; i < MAX_COUNTERS; i++) {
257  for (j = 0; j < MAX_COSQ_COUNT; j++) {
258  val[i][j] = 0;
259  }
260  }
261 
262  if ((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0))) {
263  printf("%s\n\r", example_usage);
264  return OPENNSL_E_PARAM;
265  }
266 
267  /* Initialize the system. */
268  printf("Initializing the system.\r\n");
269  rc = opennsl_driver_init((opennsl_init_t *) NULL);
270 
271  if (rc != OPENNSL_E_NONE) {
272  printf("\r\nFailed to initialize the system, rc = %d (%s).\r\n",
273  rc, opennsl_errmsg(rc));
274  return 0;
275  }
276 
277  /* Add ports to default vlan. */
278  printf("Adding ports to default vlan.\r\n");
279  rc = example_switch_default_vlan_config(unit);
280  if (rc != OPENNSL_E_NONE) {
281  printf("\r\nFailed to add default ports, rc = %d (%s).\r\n",
282  rc, opennsl_errmsg(rc));
283  }
284 
285  rc = example_port_default_config(unit);
286  if (rc != OPENNSL_E_NONE)
287  {
288  printf("\r\nFailed to apply default config on ports, rc = %d (%s).\r\n",
289  rc, opennsl_errmsg(rc));
290  }
291 
292  /* Set default BST profiles */
293  rc = example_bst_default_profile_set (unit);
294  if (rc != OPENNSL_E_NONE) {
295  printf("\r\nFailed to set default profile for BST, rc = %d (%s).\r\n",
296  rc, opennsl_errmsg(rc));
297  }
298  printf("Default BST profiles are set successfully.\r\n");
299 
300  /* Register Hw trigger callback*/
301  rc = opennsl_switch_event_register (unit,
302  example_bst_trigger_callback,
303  user_data);
304  if (rc != OPENNSL_E_NONE)
305  {
306  printf("\r\nFailed to register BST callback, rc = %d (%s).\r\n",
307  rc, opennsl_errmsg(rc));
308  return OPENNSL_E_FAIL;
309  }
310  printf("BST callback for triggers is registered.\r\n");
311 
312  /* Enabling bst module. */
313  rc = opennsl_switch_control_set(unit, opennslSwitchBstEnable, 1);
314  if (rc != OPENNSL_E_NONE) {
315  printf("\r\nFailed to Enable bst, rc = %d (%s).\r\n",
316  rc, opennsl_errmsg(rc));
317  return 0;
318  }
319  printf("BST feature is enabled.\r\n");
320 
321 
322  while (1) {
323  printf("\nUser Menu: Select one of the following options\n");
324  printf("1. Enable/Disable BST feature.\n");
325  printf("2. Display BST statistics of a port.\n");
326  printf("3. Clear BST statistics of a port.\n");
327  printf("9. Launch diagnostic shell\n");
328  printf("0. Quit the application.\n");
329 
330  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
331  {
332  printf("Invalid option entered. Please re-enter.\n");
333  continue;
334  }
335  switch(choice) {
336 
337  case 1:
338  {
339  printf("\r\nConfigure BST feature: 1-Enable, 0-Disable.\r\n");
340  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
341  {
342  printf("Invalid option entered. Please re-enter.\n");
343  continue;
344  }
345  if((choice != 0) && (choice != 1))
346  {
347  printf("Invalid input. Please re-enter.\n");
348  continue;
349  }
350 
351  rc = opennsl_switch_control_set(unit, opennslSwitchBstEnable, choice);
352  if (rc != OPENNSL_E_NONE) {
353  printf("\r\nFailed to %s BST feature, rc = %d (%s).\r\n",
354  (choice == 0)? "Disable" : "Enable", rc, opennsl_errmsg(rc));
355  return OPENNSL_E_FAIL;
356  }
357 
358  printf("BST feature is %s.\r\n", (choice == 0)? "Disabled" : "Enabled");
359  break;
360  } /* End of case 1 */
361 
362  case 2:
363  {
364  printf("\r\nEnter the port number.\r\n");
365  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
366  {
367  printf("Invalid option entered. Please re-enter.\n");
368  continue;
369  }
370  rc = opennsl_port_gport_get (unit, port, &gport);
371  if (rc != OPENNSL_E_NONE)
372  {
373  return OPENNSL_E_FAIL;
374  }
375  for (index = 0; index < MAX_COUNTERS; index++)
376  {
377  rc = (opennsl_cosq_bst_stat_sync(unit, id_list[index].bid));
378  if (rc != OPENNSL_E_NONE) {
379  printf("\r\nFailed to sync the state of port, rc = %d (%s).\r\n",
380  rc, opennsl_errmsg(rc));
381  break;
382  }
383  }
384 
385  for (index = 0; index < MAX_COUNTERS; index++)
386  {
387  for (cosq = 0; cosq <=7; cosq++)
388  {
389  rc = opennsl_cosq_bst_stat_get(unit, gport, cosq,
390  id_list[index].bid, options, &val[index][cosq]);
391  if (rc != OPENNSL_E_NONE) {
392  printf("\r\nFailed to get the port stats, rc = %d (%s).\r\n",
393  rc, opennsl_errmsg(rc));
394  break;
395  }
396  printf("BST Counter: %s for COS queue: %d is : %llu\n",
397  id_list[index].name, cosq, val[index][cosq]);
398  }
399  printf("\n");
400  }
401 
402  break;
403  } /* End of case 2 */
404 
405  case 3:
406  {
407  printf("\r\nEnter the port number.\r\n");
408  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
409  {
410  printf("Invalid option entered. Please re-enter.\n");
411  continue;
412  }
413  rc = opennsl_port_gport_get (unit, port, &gport);
414  if (rc!= OPENNSL_E_NONE)
415  {
416  return OPENNSL_E_FAIL;
417  }
418 
419  for (index = 0; index < MAX_COUNTERS; index++)
420  {
421  for (cosq = 0; cosq <= 7; cosq++)
422  {
423  rc = opennsl_cosq_bst_stat_clear(unit, gport, cosq,
424  id_list[index].bid);
425  if (rc != OPENNSL_E_NONE) {
426  printf("\r\nFailed to clear the port stats, rc = %d (%s).\r\n",
427  rc, opennsl_errmsg(rc));
428  break;
429  }
430  }
431  printf("\n");
432  }
433 
434  printf("\r\nPort %d stats cleared\r\n", port);
435  break;
436  } /* End of case 3 */
437 
438  case 9:
439  {
440  opennsl_driver_shell();
441  break;
442  }
443 
444  case 0:
445  {
446  printf("Exiting the application.\n");
447  return OPENNSL_E_NONE;
448  }
449 
450  default :
451  break;
452  } /* End of switch */
453  } /* End of while */
454 
455  return 0;
456 }
