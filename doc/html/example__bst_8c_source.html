21  * \brief OpenNSL example program for showing statistics
22  *
23  * \details OpenNSL example program for displaying and clearing BST
24  * statistics of a given interface.

27 #include <stdio.h>
28 #include <stdlib.h>
29 #include <string.h>
30 #include <sal/commdefs.h>
31 #include <opennsl/error.h>
32 #include <opennsl/cosq.h>
33 #include <opennsl/vlan.h>
34 #include <opennsl/switch.h>
35 #include <sal/driver.h>
36 #include <examples/util.h>
37 
38 #define DEFAULT_VLAN 1
39 #define MAX_COUNTERS 4
40 #define MAX_COSQ_COUNT 8
41 #define DEFAULT_UNIT 0
42 #define MAX_DIGITS_IN_CHOICE 5
43 
44 char example_usage[] =
45 "Syntax: example_bst \n\r"
46 " \n\r"
47 "Paramaters: None \n\r"
48 " \n\r"
49 "Example: The following command is used to see the bst stats of a port \n\r"
50 " example_bst \n\r"
51 " \n\r"
52 "Usage Guidelines: This program request the user to enter the port \n\r"
53 " number interactively \n\r";
54 
55 typedef struct {
56  opennsl_bst_stat_id_t bid;
57  char name[50];
58 } example_bst_counter_t;
59 /* __doxy_func_body_end__ */

61 
62 /*****************************************************************/
69 /* Main function for BST application */
70 int main(int argc, char *argv[])
71 {
72  opennsl_error_t rc;
73  int choice;
74  int unit = DEFAULT_UNIT;
75  int index,i,j;
76  opennsl_gport_t gport;
77  opennsl_port_t port;
78  opennsl_cos_queue_t cosq;
79  uint32 options = 0;
80  example_bst_counter_t id_list[MAX_COUNTERS] = {
81  {opennslBstStatIdUcast, "opennslBstStatIdUcast"},
82  {opennslBstStatIdMcast, "opennslBstStatIdMcast"},
83  {opennslBstStatIdPriGroupShared, "opennslBstStatIdPriGroupShared"},
84  {opennslBstStatIdPriGroupHeadroom, "opennslBstStatIdPriGroupHeadroom"}
85  };
86  uint64 val[MAX_COUNTERS][MAX_COSQ_COUNT];
87 
88  /* Intitializing val array */
89  for (i = 0; i < MAX_COUNTERS; i++) {
90  for (j = 0; j < MAX_COSQ_COUNT; j++) {
91  val[i][j] = 0;
92  }
93  }
94 
95  if ((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0))) {
96  printf("%s\n\r", example_usage);
97  return OPENNSL_E_PARAM;
98  }
99 
100  /* Initialize the system. */
101  printf("Initializing the system.\r\n");
102  rc = opennsl_driver_init();
103 
104  if (rc != OPENNSL_E_NONE) {
105  printf("\r\nFailed to initialize the system, rc = %d (%s).\r\n",
106  rc, opennsl_errmsg(rc));
107  return 0;
108  }
109 
110  /* Add ports to default vlan. */
111  printf("Adding ports to default vlan.\r\n");
112  rc = example_switch_default_vlan_config(unit);
113  if (rc != OPENNSL_E_NONE) {
114  printf("\r\nFailed to add default ports, rc = %d (%s).\r\n",
115  rc, opennsl_errmsg(rc));
116  }
117 
118  /* Enabling bst module. */
119  rc = opennsl_switch_control_set(unit, opennslSwitchBstEnable, 1);
120  if (rc != OPENNSL_E_NONE) {
121  printf("\r\nFailed to Enable bst, rc = %d (%s).\r\n",
122  rc, opennsl_errmsg(rc));
123  return 0;
124  }
125  printf("BST feature is enabled.\r\n");
126 
127 
128  while (1) {
129  printf("\nUser Menu: Select one of the following options\n");
130  printf("1. Display bst statistics of a port.\n");
131  printf("2. Clear bst statistics of a port.\n");
132 #ifndef CDP_EXCLUDE
133  printf("9. Launch diagnostic shell\n");
134 #endif
135  printf("0. Quit the application.\n");
136 
137  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
138  {
139  printf("Invalid option entered. Please re-enter.\n");
140  continue;
141  }
142  switch(choice) {
143 
144  case 1:
145  {
146  printf("\r\nEnter the port number.\r\n");
147  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
148  {
149  printf("Invalid option entered. Please re-enter.\n");
150  continue;
151  }
152  rc = opennsl_port_gport_get (unit, port, &gport);
153  if (rc != OPENNSL_E_NONE)
154  {
155  return OPENNSL_E_FAIL;
156  }
157  for (index = 0; index < MAX_COUNTERS; index++)
158  {
159  rc = (opennsl_cosq_bst_stat_sync(unit, id_list[index].bid));
160  if (rc != OPENNSL_E_NONE) {
161  printf("\r\nFailed to sync the state of port, rc = %d (%s).\r\n",
162  rc, opennsl_errmsg(rc));
163  break;
164  }
165  }
166 
167  for (index = 0; index < MAX_COUNTERS; index++)
168  {
169  for (cosq = 0; cosq <=7; cosq++)
170  {
171  rc = opennsl_cosq_bst_stat_get(unit, gport, cosq,
172  id_list[index].bid, options, &val[index][cosq]);
173  if (rc != OPENNSL_E_NONE) {
174  printf("\r\nFailed to get the port stats, rc = %d (%s).\r\n",
175  rc, opennsl_errmsg(rc));
176  break;
177  }
178  printf("BST Counter: %s for COS queue: %d is : %llu\n",
179  id_list[index].name, cosq, val[index][cosq]);
180  }
181  printf("\n");
182  }
183 
184  break;
185  } /* End of case 1 */
186 
187  case 2:
188  {
189  printf("\r\nEnter the port number.\r\n");
190  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
191  {
192  printf("Invalid option entered. Please re-enter.\n");
193  continue;
194  }
195  rc = opennsl_port_gport_get (unit, port, &gport);
196  if (rc!= OPENNSL_E_NONE)
197  {
198  return OPENNSL_E_FAIL;
199  }
200 
201  for (index = 0; index < MAX_COUNTERS; index++)
202  {
203  for (cosq = 0; cosq <= 7; cosq++)
204  {
205  rc = opennsl_cosq_bst_stat_clear(unit, gport, cosq,
206  id_list[index].bid);
207  if (rc != OPENNSL_E_NONE) {
208  printf("\r\nFailed to clear the port stats, rc = %d (%s).\r\n",
209  rc, opennsl_errmsg(rc));
210  break;
211  }
212  }
213  printf("\n");
214  }
215 
216  printf("\r\nPort %d stats cleared\r\n", port);
217  break;
218  } /* End of case 2 */
219 
220 #ifndef CDP_EXCLUDE
221  case 9:
222  {
223  opennsl_driver_shell();
224  break;
225  }
226 #endif
227 
228  case 0:
229  {
230  printf("Exiting the application.\n");
231  return OPENNSL_E_NONE;
232  }
233 
234  default :
235  break;
236  } /* End of switch */
237  } /* End of while */
238 
239  return 0;
240 }
